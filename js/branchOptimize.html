    代码分支优化

    # 代码分支优化

    在遇到一些逻辑判断时一般会使用 if/switch，随着分支条件的增加，会逐渐难以阅读、维护，需要进行分支优化。

    # 举个栗子

    有一个openPage方法，接收一个参数path，实现打开某个页面的功能

    1. 用 if 声明全部分支，随着页面增多，要写的 if 也越来越多
    function openPage(path) {
  if (path === "1") {
    // other works
    return console.log("即将打开页面1");
  }
  if (path === "2") {
    // other works
    return console.log("即将打开页面2");
  }
  if (path === "3") {
    // other works
    return console.log("即将打开页面3");
  }

  // default
  console.log("页面不存在");
}

openPage("1"); // => '即将打开页面1'

    1. 使用 switch，和 if 差不了多少
    function openPage(path) {
  switch (path) {
    case "1":
      // other works
      console.log("即将打开页面1");
      break;
    case "2":
      // other works
      console.log("即将打开页面2");
      break;
    case "3":
      // other works
      console.log("即将打开页面3");
      break;
    default:
      console.log("页面不存在");
      break;
  }
}

    # 开始优化

    创建一份映射表：页面路径为键，打开页面的操作为值，后续新增页面时，只需要往映射表中增加键值对即可。

    1. 在 ES6 前，可以通过 Object 模拟映射表
    function openPage(path) {
  const pageMap = {
    1: () => {
      console.log("即将打开页面1");
      // other works
    },
    2: () => {
      console.log("即将打开页面2");
      // other works
    },
    3: () => {
      console.log("即将打开页面3");
      // other works
    },
  };

  try {
    pageMap[path]();
  } catch (e) {
    console.log("页面不存在");
  }
}

openPage("1"); // => '即将打开页面1'

    1. 使用 ES6 的Map函数

      Map 和 Object 最大的区别是：Object 的键只能是字符串或者 Symbol 类型，而 Map 的键可以是任意值。

    function openPage(path) {
  const pageMap = new Map([
    [
      "1",
      () => {
        // other works
        console.log("即将打开页面1");
      },
    ],
    [
      "2",
      () => {
        // other works
        console.log("即将打开页面2");
      },
    ],
    [
      "3",
      () => {
        // other works
        console.log("即将打开页面3");
      },
    ],
  ]);

  try {
    pageMap.get(path)();
  } catch (e) {
    console.log("页面不存在");
  }
}

openPage("1");

    # 难度升级

    页面下有子级页面，通过传入第二个参数childPath跳转到某页面的某个子页面，这时Map的威力开始显现。
    由于 Map 的键可以是任意值，改造 Map 的键为对象{path: xx, childPath: xx}（后续增加其他判断条件，维护也更方便），并将函数接收的两个参数也组合成相同格式，最后进行对比匹配出对应页面。

    function openPage(path, childPath) {
  const params = {
    path,
    ...(childPath ? { childPath } : null),
  };

  const pageMap = new Map([
    [
      { path: "1" },
      () => {
        // other works
        console.log("即将打开页面1");
      },
    ],
    [
      { path: "1", childPath: "1" },
      () => {
        // other works
        console.log("即将打开页面1-子页面1");
      },
    ],
    [
      { path: "1", childPath: "2" },
      () => {
        // other works
        console.log("即将打开页面1-子页面2");
      },
    ],
    [
      { path: "2" },
      () => {
        // other works
        console.log("即将打开页面2");
      },
    ],
    [
      { path: "2", childPath: "1" },
      () => {
        // other works
        console.log("即将打开页面2-子页面1");
      },
    ],
  ]);

  try {
    let page = [...pageMap].find(
      ([key]) => JSON.stringify(key) === JSON.stringify(params)
    );
    page[1]();
  } catch (e) {
    console.log("页面不存在");
  }
}

openPage("1", "2"); // => "即将打开页面1-子页面2"
