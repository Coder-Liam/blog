    fullModal-浏览器返回兼容

    # fullModal-浏览器返回兼容

    # fullModal 介绍

    在开发后台管理系统时，存在大量列表页与详情页，列表页跳转详情页通常需要保留列表页的状态，一般会使用keep-alive缓存页面，但由于页面数量多，缓存页面的操作繁琐，且详情页需要接收的 props 比较多时，使用路由跳转容易使 url 不够纯净。

    因此可以使用fullModal达到缓存页面的效果：制作一个满屏弹窗组件，将详情页作为组件的形式引入，并传递到 fullModal 中显示，既保留列表页状态，也使得参数传递更加方便。

    # fullModal 的缺陷

    由于进入详情页实际没有路由跳转，在详情页点击浏览器后退按钮时，会发现直接从详情页跳到了列表页的上一个页面。

    # 解决思路

    historye.state中记录了当前历史栈顶部的状态值，在打开 fullModal 展示详情页时，使用history.pushState()在不刷新浏览器的情况下，创建新的状态信息到历史栈中（相当于创建了新的浏览记录）。
    最后通过popstate事件监听浏览器后退事件，判断当前是否在列表页触发的后退事件，是则隐藏 fullModal 组件。

    # 代码实现

    封装为 hook 方便复用

    import { onUnmounted } from "vue";

export function useListenerAppBack(backAction) {
  const originState = history.state.current;
  const pushHistoryState = () => {
    window.history.pushState(
      { target: originState, random: Math.random() },
      "",
      location.href
    );
  };

  const onPopstate = (e) => {
    if (e.state && e.state.current === originState) {
      backAction?.();
    }
  };

  // 监听 popstate 事件
  window.addEventListener("popstate", onPopstate, false);

  onUnmounted(() => {
    // 移除监听 popstate 事件
    window.removeEventListener("popstate", onPopstate, false);
  });

  return {
    pushHistoryState,
  };
}

    <template>
  <a-table
    size="small"
    :rowKey="(_, index) => index"
    :scroll="{ x: 1300 }"
    :pagination="pagination"
    :columns="columns"
    :data-source="tableData"
    :loading="tableLoading"
    @change="tableChange"
    @showSizeChange="tableShowSizeChange"
  >
    <template #cooperateStatus="{ record }">
      <div
        class="dot"
        :class="{
          'dot-finished': record.cooperateStatus === CooperateStatusEnums.NORMAL,
          'dot-closed': record.cooperateStatus === CooperateStatusEnums.TERMINATION
        }"
      ></div>
      <span>{{ CooperateStatusZhMap[record.cooperateStatus] }}</span>
    </template>
    <template #operate="{ record }">
      <span class="operate" @click="onShowDetail(record)">详情</span>
    </template>
  </a-table>

  <full-modal :visible="detailVisible">
    <detail
      :storeCode="detailStoreCode"
      :storeName="detailStoreName"
      :defaultCreateTime="searchForm.createTime"
      @close="detailVisible = false"
    ></detail>
  </full-modal>
</template>

<script setup name="ledgerList">
  import { ref } from 'vue'
  import fullModal from '@/components/fullModal/index.vue'
  import detail from './detail/index.vue'
  import { useListenerAppBack } from '@/hooks/useListenerAppBack'

  function onSearch() {
    // onSearch
  }

  const detailVisible = ref(false)
  const detailStoreCode = ref(null)
  const detailStoreName = ref(null)
  // 从详情页返回时进行的操作
  const backAction = () => {
    detailVisible.value = false
    onSearch()
  }
  // 添加路由历史状态
  const { pushHistoryState } = useListenerAppBack(backAction)
  // 打开详情页
  const onShowDetail = (record) => {
    detailVisible.value = true
    detailStoreCode.value = record.storeCode
    detailStoreName.value = record.storeName
    pushHistoryState()
  }
</script>
